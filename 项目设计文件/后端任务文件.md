```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id            String    @id @default(cuid())
  email         String    @unique
  passwordHash  String    @map("password_hash")
  name          String
  avatar        String?
  education     String?
  targetPosition String?  @map("target_position")
  role          String    @default("user")
  createdAt     DateTime  @default(now()) @map("created_at")
  updatedAt     DateTime  @updatedAt @map("updated_at")

  resumes       Resume[]
  interviews    Interview[]
  careerPlans   CareerPlan[]

  @@map("users")
}

model Resume {
  id          String    @id @default(cuid())
  userId      String    @map("user_id")
  title       String
  fileUrl     String?   @map("file_url")
  parsedData  Json?     @map("parsed_data")
  skills      String[]
  status      String    @default("parsing")
  createdAt   DateTime  @default(now()) @map("created_at")
  updatedAt   DateTime  @updatedAt @map("updated_at")

  user        User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  interviews  Interview[]

  @@index([userId, createdAt])
  @@map("resumes")
}

model Interview {
  id              String    @id @default(cuid())
  userId          String    @map("user_id")
  resumeId        String?   @map("resume_id")
  targetPosition  String    @map("target_position")
  difficulty      String    @default("medium")
  status          String    @default("in_progress")
  questionCount   Int       @default(0) @map("question_count")
  score           Int?
  startedAt       DateTime  @default(now()) @map("started_at")
  completedAt     DateTime? @map("completed_at")
  overallFeedback Json?     @map("overall_feedback")

  user            User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  resume          Resume?   @relation(fields: [resumeId], references: [id], onDelete: SetNull)
  messages        InterviewMessage[]

  @@index([userId, status, createdAt])
  @@index([targetPosition])
  @@map("interviews")
}

model InterviewMessage {
  id          String    @id @default(cuid())
  interviewId String    @map("interview_id")
  role        String
  content     String
  questionType String?  @map("question_type")
  feedback    String?
  createdAt   DateTime  @default(now()) @map("created_at")

  interview   Interview @relation(fields: [interviewId], references: [id], onDelete: Cascade)

  @@index([interviewId, createdAt])
  @@map("interview_messages")
}

model CareerPlan {
  id              String    @id @default(cuid())
  userId          String    @map("user_id")
  targetPosition  String    @map("target_position")
  currentSkills   String[]  @map("current_skills")
  gapSkills       String[]  @map("gap_skills")
  roadmap         Json
  marketInsight   Json?     @map("market_insight")
  progress        Int       @default(0)
  createdAt       DateTime  @default(now()) @map("created_at")
  updatedAt       DateTime  @updatedAt @map("updated_at")

  user            User      @relation(fields: [userId], references: [id], onDelete: Cascade)
```

### T-005 JWT 认证逻辑（注册/登录/Token 刷新）

#### 实现文件及职责

##### `src/auth/auth.controller.ts`

| 端点 | 方法 | 说明 |
|------|------|------|
| `POST /auth/register` | `register()` | 用户注册 |
| `POST /auth/login` | `login()` | 用户登录 |
| `POST /auth/refresh` | `refresh()` | 刷新 Token |
| `GET /auth/profile` | `getProfile()` | 获取当前用户信息 |
| `PATCH /auth/profile` | `updateProfile()` | 修改个人资料 |

##### `src/auth/auth.service.ts`

**注册逻辑：**

1. 校验邮箱是否已注册（`Prisma.user.findUnique({ where: { email } })`）
2. 已注册抛 409 Conflict 异常
3. bcrypt.hash(password, 10) 加密密码
4. Prisma.user.create() 创建用户
5. 生成 Access Token（15min）和 Refresh Token（7d）
6. 返回用户信息 + Token 对

**登录逻辑：**

1. 根据 email 查找用户
2. 不存在抛 401 Unauthorized
3. bcrypt.compare() 校验密码
4. 校验失败抛 401
5. 生成 Token 对并返回

**Token 刷新逻辑：**

1. 校验 refreshToken 是否有效（JWT verify）
2. 从 payload 中提取 userId
3. 生成新的 Access Token 和 Refresh Token
4. 可选：旧 Refresh Token 加入黑名单（Redis）

##### `src/auth/strategies/jwt.strategy.ts`

```typescript
@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor() {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      secretOrKey: process.env.JWT_SECRET,
    });
  }

  async validate(payload: { sub: string; email: string }) {
    // 返回用户 ID 和邮箱，附加到 request.user
    return { id: payload.sub, email: payload.email };
  }
}
```

| 字段 | 内容 |
|------|------|

**流程：**

```
Bull 队列消费 → 读取文件 → 提取文本 → LLM 结构化提取 → 更新数据库
```

**详细步骤：**

#### `resume.parser.ts` — 文件文本提取

```typescript
@Injectable()
export class ResumeParser {
  async extractText(filePath: string): Promise<string> {
    const ext = path.extname(filePath).toLowerCase();
    if (ext === '.pdf') {
      // 使用 pdf-parse 提取 PDF 文本
      const dataBuffer = fs.readFileSync(filePath);
      const pdfData = await pdfParse(dataBuffer);
      return pdfData.text;
    } else if (ext === '.docx') {
      // 使用 mammoth 提取 Word 文本
      const result = await mammoth.extractRawText({ path: filePath });
      return result.value;
    }
    throw new BadRequestException('不支持的文件格式');
  }

  async parseWithLLM(text: string): Promise<ParsedResumeData> {
    // 调用 AI 服务，传入 System Prompt + 简历文本
    // 返回结构化 JSON
    return this.aiService.parseResume(text);
  }
}
```

#### `resume.processor.ts` — Bull 队列消费者

```typescript
@Processor('resume-parser')
export class ResumeProcessor {
  @Process()
  async handleParsing(job: Job<{ resumeId: string }>) {
    const { resumeId } = job.data;
    try {
      // 1. 从 DB 获取简历记录
      const resume = await this.prisma.resume.findUnique({ where: { id: resumeId } });
      // 2. 提取文本
      const text = await this.resumeParser.extractText(resume.fileUrl);
      // 3. LLM 结构化提取
      const parsedData = await this.resumeParser.parseWithLLM(text);
      // 4. 更新数据库
      await this.prisma.resume.update({
        where: { id: resumeId },
        data: {
          parsedData: parsedData as any,
          skills: parsedData.skills || [],
          status: 'completed',
        },
      });
    } catch (error) {
      // 解析失败更新状态
      await this.prisma.resume.update({
        where: { id: resumeId },
        data: { status: 'failed' },
      });
    }
  }
}
```

## 十三、开发注意事项

### 13.1 错误处理

- 所有 Service 层抛出自定义 HttpException（含状态码和中文错误消息）
- 全局异常过滤器捕获并统一格式返回
- WebSocket 异常通过 `ws-exception.filter.ts` 处理

### 13.2 数据校验

- 所有 DTO 使用 `class-validator` 装饰器（如 `@IsEmail`, `@MinLength`, `@IsOptional`）
- 全局 `ValidationPipe` 注册在 `main.ts` 中
- 敏感信息（密码哈希）永远不在日志中输出

### 13.3 安全

- 密码 bcrypt 加密（saltRounds = 10）
- JWT Token 过期时间：Access 15 分钟，Refresh 7 天
- 所有需要认证的接口使用 `@UseGuards(JwtAuthGuard)` 保护
- Refresh Token 使用后可轮换（旧 token 加入 Redis 黑名单）
- 文件上传做类型校验（防止恶意文件)

### 13.4 性能

- 列表接口强制分页（最大 pageSize 50）
- JSONB 查询使用 `@>` 操作符走 GIN 索引
- LLM 调用设置超时（默认 30s，流式模式 60s）
- 简历解析使用异步队列处理，不阻塞 HTTP 响应
- WebSocket 连接做好心跳检测和断线重连

### 13.5 AI 服务配置

```typescript
// AI Provider 选择逻辑

  ? new OpenAIProvider()
  : process.env.DASHSCOPE_API_KEY
    ? new DashScopeProvider()
```
